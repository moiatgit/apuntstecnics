Introducció

Aquest document té com a objectiu oferir l'estudiant una primera impressió d'un programa en java tot comparant-lo amb una versió del programa en un altre llenguatge conegut: el C

Per a aprofitar al màxim aquesta activitat, caldria tenir alguns coneixements bàsics del llenguatge C (e.g. directius de precompil·lació, funcions)

#include <stdio.h>
#include <math.h>
#define PI 3.14159265358979323846264338327

/* calcula l'àrea d'un cercle a partir del seu radi */
double calcula_area_cercle(double radi) {
    return PI * pow(radi, 2);
}

int main() {
  double radi, area;
  printf("Càlcul de l'àrea d'un cercle\n");
  printf("Introduïu el radi: ");
  scanf("%lg", &radi);
  area = calcula_area_cercle(radi);
  printf("L'àrea és %.16lg\n", area);
  return 0;
}

import java.io.*;

public class CalculaRadi{
  private static double calcula_area(double radi) {
      return Math.PI * Math.pow(radi, 2);
  }
  public static void main(String[] args) throws IOException {
    System.out.println("Càlcul de l'àrea d'un cercle");
    System.out.print("Introduïu el radi: ");
    BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
    String linia = reader.readLine();
    double radi = Double.parseDouble(linia);  // conversió de String a double
    double area = calcula_area(radi);
    System.out.println(" L'àrea és " + area);
  }
}
